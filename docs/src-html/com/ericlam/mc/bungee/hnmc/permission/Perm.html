001package com.ericlam.mc.bungee.hnmc.permission;
002
003
004import net.md_5.bungee.api.CommandSender;
005
006import java.util.ArrayList;
007import java.util.List;
008
009public class Perm {
010    public static final String ADMIN = "hypernite.admin";
011    public static final String HELPER = "hypernite.helper";
012    public static final String MOD = "hypernite.mod";
013    public static final String BUILDER = "hypernite.builder";
014    public static final String DONOR = "hypernite.donor";
015    public static final String OWNER = "*";
016
017    public static boolean hasPermission(CommandSender sender, String permission) {
018        boolean reuslt = false;
019        if (permission == null) return true;
020        for (String Pnode : getPermHierarchy(permission)) {
021            reuslt = reuslt || sender.hasPermission(Pnode);
022        }
023        return reuslt;
024    }
025
026    public static List<String> getPermHierarchy(String string) {
027        String[] node = string.split("\\.");
028        List<String> permissionNode = new ArrayList<>();
029        StringBuilder stringBuilder = new StringBuilder();
030        for (int i = 0; i < node.length; i++) {
031            stringBuilder.append(node[i]);
032            permissionNode.add(stringBuilder.toString());
033            if (i != node.length - 1) stringBuilder.append(".");
034        }
035        return permissionNode;
036    }
037}
